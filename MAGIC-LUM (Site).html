           try {

               const response = await fetch(endpoint, {

                   method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload)

           } catch (error) { console.error(error); return null; }

       }

               const data = await fetchGemini(payload);

       let annualLoss = 0;

       function calculateROI() {

           const costPaint = 15000; 

           const costAdhesive = 25000; 

           const costBanner = 45000; 

           const costSign = 120000; 


           let totalTraditionalCost = 0;


           if(document.getElementById('checkPintura').checked) totalTraditionalCost += (parseFloat(document.getElementById('valPintura').value) || 0) * costPaint * 4; 

           if(document.getElementById('checkAdhesivos').checked) totalTraditionalCost += (parseFloat(document.getElementById('valAdhesivos').value) || 0) * costAdhesive * 2; 

           if(document.getElementById('checkPendones').checked) totalTraditionalCost += (parseFloat(document.getElementById('valPendones').value) || 0) * costBanner * 1; 

           if(document.getElementById('checkLetreros').checked) totalTraditionalCost += (parseFloat(document.getElementById('valLetreros').value) || 0) * costSign * 1;


           // Logic: 5 Year projection now

           const projectedCost5Years = totalTraditionalCost * 5;

           const investment = Math.max(2450000, totalTraditionalCost * 0.4); // Scalable investment logic

           

           const savings5Years = projectedCost5Years - (investment * 1.2); // Investment + 20% maintenance over 5 years

           const roiMonths = (investment / (totalTraditionalCost/12)).toFixed(1);

           

           const fmt = new Intl.NumberFormat('es-CL', { style: 'currency', currency: 'CLP', maximumFractionDigits: 0 });

           

           document.getElementById('resInversion').innerText = fmt.format(investment);

           document.getElementById('resAhorroTotal').innerText = fmt.format(savings5Years);

           document.getElementById('resMeses').innerText = roiMonths + " m";

           

           document.getElementById('resResiduos').innerText = Math.round(totalTraditionalCost * 0.00005) + " kg";

           annualLoss = totalTraditionalCost;

       }


       let currentLoss = 23287;

       setInterval(() => {

           if(annualLoss > 0) {

               const lossPerSecond = annualLoss / 31536000; 

               currentLoss += lossPerSecond * 0.1; 

               document.getElementById('lossCounter').innerText = "$" + Math.floor(currentLoss).toLocaleString('es-CL');

           }

       }, 100);
